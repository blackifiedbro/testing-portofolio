    -- Combat System
    local cooldown = false
     
    function attack(player)
        if not cooldown then
            cooldown = true
            playAnimation(player, "Slash")
            applyDamage(player, 25)
     
            task.delay(0.4, function()
                cooldown = false
            end)
        end
    end
